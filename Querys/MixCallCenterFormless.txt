use
	mediaccess
select
	A.CodProducto,
	EMP.NOMBREFISCAL,
	B.CODAFILIADO,
	A.APELLIDOPATERNO+' '+A.APELLIDOMATERNO+' '+A.NOMBRES,
	P.PARENTESCO,
	E.ESTADO,
	M.MOTIVOLLAMADA,
	B.FECHALLAMADA,
	E.ESTADO
from	
	bandejaentrada B 
	INNER JOIN iss..afiliado A ON A.CODAFILIADO=B.CODAFILIADO AND A.CODEMPRESA=B.CODEMPRESA AND A.CORRELATIVO=B.CORRELATIVO
	INNER JOIN ESTADO E ON E.CODESTADO=B.CODESTADO
	INNER JOIN MOTIVOLLAMADA M ON M.CODMOTIVOLLAMADA=B.CODMOTIVOLLAMADA
	INNER JOIN TIPOLLAMADA T ON T.CODTIPOLLAMADA=B.CODTIPOLLAMADA
	INNER JOIN SERVICIO S ON S.CODSERVICIO=B.CODSERVICIO
	INNER JOIN PARENTESCO P ON P.CODPARENTESCO=A.CODPARENTESCO
	left join iss.saMediaccess.empresasPyme py on a.codpymecolectivo = py.codpyme AND A.CODEMPRESA=PY.CODEMPRESA
	INNER JOIN ISS..EMPRESA EMP ON EMP.CODEMPRESA=A.CODEMPRESA
where 
	B.CODAFILIADO <>'0'
	and A.CODEMPRESA in ('384')
	and B.FechaLlamada between '2015/12/16' and GETDATE() 
order by 
	b.fechallamada asc
